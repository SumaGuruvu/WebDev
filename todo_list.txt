    <script>
        function addTaskToList(){
            
                var description = document.getElementById("text");
                var date = document.getElementById("dateValue");
               
                if(description.value!=""&&date.value!=""){
                var table = document.getElementById("tab");
                var row = table.insertRow();
                var datecell = row.insertCell(0);
                var descCell = row.insertCell(1);
                var statusCell = row.insertCell(2);
    
                datecell.innerHTML= date.value;
                descCell.innerHTML = description.value;
                statusCell.innerHTML = '<input type="checkbox"/>';  
    
                document.getElementById("text").value = "";
                document.getElementById("dateValue").value = "";
            }else{
                alert("fill the required fields");
            }

            
        }
    </script>
